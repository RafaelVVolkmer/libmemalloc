218static inline int LOG_output(log_level_t level,
219 const char *color,
220 const char *prefix,
221 const char *file,
222 const char *func,
223 int line,
224 const char *fmt,
225 ...)
226{
227 int ret = EXIT_SUCCESS;
228
229 FILE *out = NULL;
230
231 static pthread_mutex_t log_mutex = PTHREAD_MUTEX_INITIALIZER;
232
233 struct timespec ts;
234 struct tm *ptm = NULL;
235
236 time_t now = 0;
237
238 if (level > LOG_LEVEL)
239 {
240 ret = -EIO;
241 goto function_output;
242 }
243
244 pthread_mutex_lock(&log_mutex);
245
246 memset(&ts, 0, sizeof(ts));
247
248 clock_gettime(CLOCK_REALTIME, &ts);
249
250 now = ts.tv_sec;
251 ptm = localtime(&now);
252
253 out = (level <= LOG_LEVEL_WARNING) ? stderr : stdout;
254 LOG_PRINTF(out,
255 "[%02d:%02d:%02d.%03ld] ",
256 ptm->tm_hour,
257 ptm->tm_min,
258 ptm->tm_sec,
259 ts.tv_nsec / 1'000'000u);
260
261 if (isatty(fileno(out)))
262 LOG_PRINTF(out, "%s%s%s ", color, prefix, COLOR_RESET);
263 else
264 LOG_PRINTF(out, "%s ", prefix);
265
266 va_list args;
267 va_start(args, fmt);
268 LOG_VPRINTF(out, fmt, args);
269 va_end(args);
270
271 LOG_PRINTF(out, " (at %s:%d:%s())\n", file, line, func);
272 pthread_mutex_unlock(&log_mutex);
273
274function_output:
275 return ret;
276}
